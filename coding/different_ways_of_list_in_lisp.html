Different Ways to Say '(I LOVE YOU) in LISP

All runnable on GNU CLISP 2.49.92. "Compiled" from "99 ways to say '(I love you)"(in Racket) by Prof Matt Might.

;;;
'(I love you)

;;;
(quote (I love you))

;;;
(list 'I 'love 'you)

;;;
(list (quote I) (quote love) (quote you))

;;;
(cons 'I (cons 'love (cons 'you '())))

;;;
(cons 'I (cons 'love (list 'you)))

;;;
(cons 'I (list 'love 'you))

;;;
(cons 'I '(love you))

;;;
'(I love . (you))

;;;
'(I . (love . (you)))

;;;
'(I . (love . (you . ())))

;;;
'(I . (love you))


;;;
`(I love you)

;;;
`(I ,'love you)

;;;
`(I ,`love you)

;::
(let ((verb 'love))
  `(I ,verb you))

;;;
`(I love . (you))

;;;
`(I love . ,(list 'you))

;;;
`(I love ,@'(you))

;;;
`(I ,@(list 'love 'you))

;;;
`(,@(list 'I 'love) you)

;;;
`(,@'(I love you))

;;;
`,'(I love you)

;;;
`(I love you . ,'())

;;;
(car (list '(I love you)))

;;;
(cdr '(Hark! I love you))

;;;
(let ()
  '(I love you not)
  '(I love you))

;;;
((lambda (a c b) (list a b c)) 'I 'you 'love)

;;;
(apply (lambda (a c b) (list a b c)) 
       (list 'I 'you 'love))    

;;;
(append '(I love) '(you))

;;;
(append '(I) '(love) '(you))

;;;
(reverse '(you love I))

;;;
(remove 'cannot '(I cannot love you))

;;;
(remove-duplicates '(I love love love you))]
  
;;;
(cond
  ((evenp 3) '(Not me))
  ((oddp  3) '(I love you)))
  
;;;
(case 1
  ((a b c) '(Not me))
  ((3 2 1) '(I love you)))


;;;
(let ((word 'love))
  (setf word 'love)
  `(I ,word you))
